CURD 神器 GORM 包介绍及实战

1 基础知识#

  • GORM 使用模型（Models）来映射一个数据库表。
  • 默认情况下，使用 ID 作为主键，使用结构体名的 snake_cases 作为表名，使用字段名的 snake_case 作为列名。
  • 并使用 CreatedAt、UpdatedAt、DeletedAt 字段追踪创建、更新和删除时间。

实际开发中建议直接指定字段名和表名。

type Animal struct {
    AnimalID int64     `gorm:"column:animalID;primarykey"` // 将列名设为 `animalID`
    Birthday time.Time `gorm:"column:birthday"`            // 将列名设为 `birthday`
    Age      int64     `gorm:"column:age"` // 将列名设为 `age`
}

func (a *Animal) TableName() string {
    return "animal"
}

2.2 原生 sql 查找一个数据#

type Result struct {
    ID   int
    Name string
    Age  int
}

var result Result
db.Raw("SELECT id, name, age FROM users WHERE name = ?", 3).Scan(&result)

2.3 原生 sql 查找多个数据#

type Result struct {
    ID   int
    Name string
    Age  int
}

res := make([]*Result,0)
db.Raw("SELECT id, name, age FROM users WHERE name > ?", 3).Scan(&res)

3 添加数据#

3.1 添加一条数据#

user := User{
	Name:      "Jinzhu3",
	Password:  "123",
	Phone:     "1",
	CreatedAt: time.Now(),
}
result := db.Create(&user) // 通过数据的指针来创建
log.Printf("result: id: %v err: %s result.RowsAffected: %v", user.ID, result.Error, result.RowsAffected)
  • user.ID：返回插入数据的主键，这个是直接赋值给 user 变量。
  • result.Error：返回 error。
  • result.RowsAffected：返回插入记录的条数。

3.2 可以批量插入多条数据#

var users = []User{{Name: "jinzhu7"}, {Name: "jinzhu8"}, {Name: "jinzhu9"}}
db.Create(&users)
for _, user := range users {
	log.Printf("id: %v, name: %v",user.ID,user.Name)
}

4 删除数据#

4.1 gorm 支持两种删除方式#

软删除

软删除是指执行 Delete 时，记录不会被从数据库中真正删除。GORM 会将 DeletedAt 设置为当前时间，并且不能通过正常的方式查询到该记录。 如果模型包含了一个 gorm.DeletedAt 字段，GORM 在执行删除操作时，会软删除该记录。

// UPDATE users SET deleted_at="2013-10-29 10:23" WHERE age = 20;
db.Where("age = ?", 20).Delete(&User{})

// SELECT * FROM users WHERE age = 20 AND deleted_at IS NULL;
db.Where("age = 20").Find(&user)

GORM 并没有真正把记录从数据库删除掉，而是只更新了 deleted_at 字段。在查询时，GORM 查询条件中新增了AND deleted_at IS NULL条件，所以这些被设置过 deleted_at 字段的记录不会被查询到。 对于一些比较重要的数据，我们可以通过软删除的方式删除记录，软删除可以使这些重要的数据后期能够被恢复，并且便于以后的排障。

可以通过下面的方式查找被软删除的记录：

// SELECT * FROM users WHERE age = 20;
db.Unscoped().Where("age = 20").Find(&users)

永久删除

如果想永久删除一条记录，可以使用 Unscoped：

// DELETE FROM orders WHERE id=10;
db.Unscoped().Delete(&order)

4.2 删除一条语句#

// delete from user where name = "Jinzhu"
// UPDATE `user` SET `deletedAt`='2022-06-16 22:06:59.383' WHERE name = 'Jinzhu' AND `user`.`deletedAt` IS NULL 
db.Where("name = ?", "Jinzhu").Delete(&User{})

5 更新数据#

5.1 更新单条数据#

user := &User{}
// SELECT * FROM `user` WHERE `user`.`deletedAt` IS NULL ORDER BY `user`.`id` LIMIT 1
db.First(&user)
user.Phone = "1000"
// UPDATE `user` SET `name`='',`password`='',`phone`='1000',`createdAt`='2022-06-18 22:34:53',`updatedAt`='2022-06-18 22:35:15.23',`deletedAt`=NULL WHERE `user`.`deletedAt` IS NULL AND `id` = 5
db.Save(user)

通过 Save 方法，可以把 user 变量中所有跟数据库不一致的字段更新到数据库中。 先获取某个资源的详细信息，再通过 user.Phone = "1000" 这类赋值语句，对其中的一些字段重新赋值。最后，调用 Save 方法更新这些字段。

也可以采用如下方法更新:

name := "xiaoming"
// [8.488ms] [rows:1] UPDATE `user` SET `password`='123456',`updatedAt`='2022-07-19 22:34:27.834' WHERE name = 'xiaoming' AND `user`.`deletedAt` IS NULL
db.Debug().Model(&User{}).Where("name = ?", name).Update("password", "123456")

5.2 更新多条语句#

phone := 1
// [8.961ms] [rows:2] UPDATE `user` SET `password`='1',`phone`='2',`updatedAt`='2022-07-19 22:40:02.632' WHERE phone = 1 AND `user`.`deletedAt` IS NULL
db.Debug().Model(&User{}).Where("phone = ?", phone).Updates(User{Password: "1", Phone: "2"})

6 查询数据#

6.1 获取第一条记录(主键升序)#

var u User
phone := "2"
//[7.439ms] [rows:1] SELECT * FROM `user` WHERE phone = '2' AND `user`.`deletedAt` IS NULL ORDER BY `user`.`id` LIMIT 1
result := db.Debug().Where("phone = ?", phone).First(&u)

log.Println("result.RowsAffected", result.RowsAffected)  // 找到的记录数
log.Println("result error", result.Error) // returns error

6.2 获取最后一条记录(主键降序)#

var u User
phone := "2"
//[7.439ms] [rows:1] SELECT * FROM `user` WHERE phone = '2' AND `user`.`deletedAt` IS NULL ORDER BY `user`.`id` DESC LIMIT 1
result := db.Debug().Where("phone = ?", phone).Last(&u)

7 事务#

func updateVirtualTx(localSql, businessSql string) (err error) {
// 开始事务
    tx := db.Begin()
	defer func() {
        if err != nil {
            fabricLog.Println("err", err.Error())
            tx.Rollback()
        } else {
        // 否则，提交事务
         tx.Commit()
        }
    }()

    // 更新本地数据
    sqlRes := tx.Debug().Exec(localSql)
    err = sqlRes.Error
    if err != nil {
        return err
    }

    // 判断影响结果这个看情况可要可不要
    if sqlRes.RowsAffected > 1 {
        err = fmt.Errorf("sqlRes.RowsAffected > 1 localsql: %s", localSql)
        return
    }

    // 更新业务数据
    businessSqlRes := tx.Debug().Exec(businessSql)
    err = businessSqlRes.Error
    if err != nil {
        return err
    }
    if businessSqlRes.RowsAffected > 1 {
        err = fmt.Errorf("businessSqlRes.RowsAffected > 1 businessSql: %s", businessSql)
        return
    }

    return
}
